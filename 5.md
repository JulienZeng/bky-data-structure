SqQueue Q;

int k=0;
while(!QueueEmpty(Q)) {
  T=Q.base[Q.front];
  if (T->lchild!=NULL)
    EnQueue(Q,T->lchild);
  if (T->rchild!=NULL)
    EnQueue(Q,T->rchild);
  DeQueue(Q,T);
  Lever[k++]=T->data;
}
Lever[k]='\0';
