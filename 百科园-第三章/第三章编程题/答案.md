Status Push(SqStack &S, SElemType e) {
	// 插入元素e为新的栈顶元素

	/**********Begin**********/
	if(S.top-S.base<S.stacksize-1) {
		*(S.top++)=e;
			return OK;
	}
	else
		return OVERFLOW; 
		 /**********  End  **********/	
}

//算法3.3　顺序栈的出栈
Status Pop(SqStack &S, SElemType &e) {
	//删除S的栈顶元素，用e返回其值	
	/**********Begin**********/
	if (S.base == S.top)
		return ERROR;//栈空
	e = *(--S.top); //栈顶指针减1，将栈顶元素赋给e
	return OK;

		 /**********  End  **********/	
}
